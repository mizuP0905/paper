\textbf{1. Hourglass geometry.}
\]

\[

\qquad
\rho:[-T_\ast,T_\ast]\to(0,\infty),
\qquad
\rho(-t)=\rho(t),
\qquad
\rho \text{ Lipschitz},

\left\{

\[

\[
\textbf{9. Casimir flux identity.}
\]

\[
C_\rho
:=
\sum_{a=1}^{\dim\mathfrak g}
d\rho(X^a)^\ast d\rho(X^a).
\]

\[
d\rho(X^a)^\ast=-d\rho(X^a).
\]

\[
d\rho(X^a)^\ast d\rho(X^a)\ge0.
\]

\[
C_\rho\ge0.
\]

\[
\rho(g)d\rho(X^a)\rho(g)^{-1}
=
d\rho(\operatorname{Ad}_gX^a).
\]

\[
\sum_a
d\rho(\operatorname{Ad}_gX^a)^\ast
d\rho(\operatorname{Ad}_gX^a)
=
\sum_a
d\rho(X^a)^\ast d\rho(X^a).
\]

\[
\rho(g)C_\rho\rho(g)^{-1}
=
C_\rho.
\]

\[
C_\rho\rho(g)=\rho(g)C_\rho
\qquad
\forall g\in G.
\]

\[
\rho\ \text{irreducible}
\quad\Longrightarrow\quad
C_\rho=C_2(\rho)I_{V_\rho}.
\]

\[
C_2(\rho)\ge0.
\]

\[
C_2(\rho)=0
\quad\Longrightarrow\quad
\sum_a\|d\rho(X^a)v\|^2=0
\quad
\forall v\in V_\rho.
\]

\[
\Longrightarrow\quad
d\rho(X^a)v=0
\quad
\forall a,\ \forall v\in V_\rho.
\]

\[
\Longrightarrow\quad
d\rho(X^a)=0
\quad
\forall a.
\]

\[
\Longrightarrow\quad
d\rho=0.
\]

\[
d\rho\neq0
\quad\Longrightarrow\quad
C_2(\rho)>0.
\]

\[
\sum_{a=1}^{\dim\mathfrak g}
\|(d\rho(X^a)\otimes I\otimes I)\Psi\|^2
=
\left\langle
\Psi,
\left(
\sum_a d\rho(X^a)^\ast d\rho(X^a)
\otimes I\otimes I
\right)\Psi
\right\rangle.
\]

\[
=
\langle\Psi,(C_2(\rho)I\otimes I\otimes I)\Psi\rangle.
\]

\[
=
C_2(\rho)\|\Psi\|^2.
\]

\[
\text{The orientation of }S_{t,s}\text{ is chosen so that the enclosed endpoint charge is the left }
V_\rho\text{ factor.}
\]

\[
\text{With this convention, the Gauss-law flux through }S_{t,s}
\text{ acts as }
d\rho(X^a)\otimes I\otimes I.
\]

\begin{center}
\parbox{0.9\linewidth}{
If the opposite orientation is chosen, the sign changes, but the Casimir norm-square estimate is unchanged.
}
\end{center}

\[
\widehat{\mathcal J}_t^a(s):\mathscr D_t\to\mathcal K_{R,\rho}(t).
\]

\[

=

\qquad
(\Psi\in\mathscr D_{\Lambda,t}^{\mathrm{phys}})
\]

\[

\]

\[
\sum_a
\|\widehat{\mathcal J}_t^a(s)\Psi\|^2
=
\sum_a

\]

\[
=
C_2(\rho)\|\Psi\|^2.
\]

\[
C_2(\rho)\|\Psi\|^2
\le
|S_{t,s}|
\left\langle
\Psi,
\left(

\right\rangle
\ge

\int_{S_{t,s}}

\ge
